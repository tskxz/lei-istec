using System;
using System.Collections.Generic;
using System.Text;

namespace Simulador.Models
{
    public class Simulador
    {
        private decimal _valorInicial = 100; // mínimo 100, máximo 50 000
        private int _nrMeses = 6; // mínimo 6, máximo 36
        private decimal _txJuros; // 10% ao ano se idade < 50 , 20% ao ano se for >= 50
        private int _anoNascimento = 1940; // !menos de 1940, n pode pedir empréstimo menor de 18
        private decimal _mensalidade;

        public Simulador()
        {
            setValorInicial(100);
            setNrMeses(6);
            setAnoNascimento(1940);
        }

        public void setValorInicial(decimal valorInicial) {
            if(valorInicial < 100)
            {
                _valorInicial = 100;
            }
            if(valorInicial > 50000) {
                _valorInicial = 50000;
            }
            _valorInicial = valorInicial;
        }

        public decimal getValorInicial() {
            return _valorInicial;
        }

        public void setNrMeses(int meses)
        {
            if(meses < 6) {
                _nrMeses = 6;
            }
            if(_nrMeses > 36)
            {
                _nrMeses = 36;
            }
            _nrMeses = meses;
        }

        public int getNrMeses() {
            return _nrMeses;
        }

        public void setTxJuros() {
            int idade = DateTime.Now.Year - _anoNascimento;
            if (idade < 50)
            {
                _txJuros = 0.10M;
            }
            if(idade > 50)
            {
                _txJuros = 0.20M;
            }
        }

        public decimal getTxJuros() {
            return _txJuros;
        }

        public void setAnoNascimento(int anoNascimento) {
            _anoNascimento = anoNascimento;
            if (_anoNascimento < 1940) {
                _anoNascimento = 1940;
            }
            if(idade() < 18) {
                Console.WriteLine("Nao pode pedir emprestimo para menores de 18 anos");
            }
            _anoNascimento = anoNascimento;
        }

        public int getAnoNascimento() {
            return _anoNascimento; 
        }

        public decimal getMensalidade()
        {
            return _mensalidade;
        }

        public int idade()
        {
            return DateTime.Now.Year - _anoNascimento;
        }

        public decimal CalcularMensalidade()
        {
            decimal valorFinal = _valorInicial * (1 + CalcularJuros());
            _mensalidade = valorFinal / _nrMeses
;            return _mensalidade;
        }

        public decimal CalcularJuros()
        {
            return _txJuros * (_nrMeses / 12M);
        }

        public void ExibirDados()
        {
            Console.WriteLine($"Valor Inicial: {_valorInicial}");
            Console.WriteLine($"Número de Meses: {_nrMeses}");
            Console.WriteLine($"Taxa de Juros: {_txJuros * 100}% ao ano");
            Console.WriteLine($"Ano de Nascimento: {_anoNascimento}");
            Console.WriteLine($"Idade: {idade()} anos");
            Console.WriteLine($"Mensalidade: {_mensalidade}");
        }
    }
}
